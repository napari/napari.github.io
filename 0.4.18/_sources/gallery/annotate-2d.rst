    from skimage import data

    print("click to add points; close the window when finished.")

    viewer = napari.view_image(data.astronaut(), rgb=True)
    points = viewer.add_points(np.zeros((0, 2)))
    points.mode = 'add'

    if __name__ == '__main__':
